# By passing in the axis = 1 argument, we can use the DataFrame.apply method
# to iterate rows instead of columns
def which_class(row):
    pclass = row['Pclass']
    if pd.isnull(pclass):
        return 'Unknown'
    elif pclass == 1:
        return 'First Class'
    elif pclass == 2:
        return 'Second Class'
    elif pclass == 3:
        return 'ThirdClass'
classes = titanic_survival.apply(which_class, axis=1)
print(classes[:10])

def is_minor(row):
    if row['Age'] < 18:
        return True

def generate_age_label(row):
    age = row['Age']
    if pd.isnull(age):
        return 'Unknown'
    elif age < 18:
        return 'minor'
    else:
        return 'adult'
age_label = titanic_survival.apply(generate_age_label, axis=1)
