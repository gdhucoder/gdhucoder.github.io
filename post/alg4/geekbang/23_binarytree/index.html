  /**
   * preorder
   */
  public static void preOrder(Node n) {
    if (n == null) {
      return;
    }
    StdOut.println(n);
    preOrder(n.left);
    preOrder(n.right);
  }

  /**
   * inorder
   */
  public static void inOrder(Node n) {
    if (n == null) {
      return;
    }
    inOrder(n.left);
    StdOut.println(n);
    inOrder(n.right);
  }

  /**
   * postorder
   */
  public static void postOrder(Node n) {
    if (n == null) {
      return;
    }
    postOrder(n.left);
    postOrder(n.right);
    StdOut.println(n);
  }
